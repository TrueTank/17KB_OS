#!/bin/bash
select var in "Текущий пользователь" "Объем используемой памяти" "Объем дискового пространства" "Запущенные процессы" "Процессы,запущенные пользователем" "Системные дата и время" "Время запуска системы" "Выход"
do
echo $var;
case "$var" in
"Текущий пользователь") whoami;;
"Объем используемой памяти") free -m;; 
"Объем дискового пространства") df --total;;
"Запущенные процессы") ps -e;;
"Процессы,запущенные пользователем") n=` who|cut -d' ' -f1|sort|uniq `; ps -u $n;;
"Системные дата и время") date -u;;
"Время запуска системы") uptime;;
"Выход") exit;;
esac
done
